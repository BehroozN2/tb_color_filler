Examples

Random

This returns one of the four possible moves randomly.

function (x, y, turn, memory) {
    let moves = ['top', 'bottom', 'left', 'right'];
    return moves[Math.floor(Math.random() * moves.length)];
}

Hard Coded

This has a set of hard coded moves for first 62 turns.

function (x, y, turn, memory) {
    if (turn <= 20) {
        return 'right';
    } else if (turn === 21) {
        return 'bottom';
    } else if (turn <= 41) {
        return 'left';
    } else if (turn === 42) {
        return 'bottom';
    } else if (turn <= 62) {
        return 'right';
    }

    // ... etc ...
}

Loop Around

This rotates around the board.

function (x, y, turn, memory) {
    if (y === 1 && x !== 20) {
        return 'right';
    } else if (x === 1) {
        return 'top';
    } else if (y === 20) {
        return 'left';
    } else if (x === 20) {
        return 'bottom';
    }
}

Horizontal

This goes left and right.

function (x, y, turn, memory) {
    if (x === 1) {
        return memory.direction = 'right';
    } else if (x === 20) {
        return memory.direction = 'left';
    } else {
        return memory.direction;
    }
}

Vertical

This goes up and down.

function (x, y, turn, memory) {
    if (y === 1) {
        return memory.direction = 'bottom';
    } else if (y === 20) {
        return memory.direction = 'top';
    } else {
        return memory.direction;
    }
}
